function getFormattedDate(dateString, lang) {
  const dateOptions = { day: "numeric", month: "long", year: "numeric", weekday: "long" };
  const date = new Date(dateString); 
  return date.toLocaleDateString(lang === "en" ? "en-US" : "tr-TR", dateOptions);
}

const LeafCard = ({ children, onNext, btnText, step, theme, showSettings, setShowSettings, settingsProps }) => {
  const contentRef = useRef(null);
  
  const getThemeColors = () => {
    if (theme === 'sepia') return 'bg-[#e3d5c6]'; 
    if (theme === 'dark') return 'bg-[#1c1917]'; 
    return 'bg-[#fffdf9]'; 
  };

  useEffect(() => { 
    if (contentRef.current) contentRef.current.scrollTo(0, 0); 
  }, [step]);

  return (
    <div className={`flex flex-col h-full relative transition-colors duration-700 overflow-hidden ${getThemeColors()}`}>
      {/* Ayarlar Butonu Her Sayfada Olsun */}
      <div className="absolute top-4 right-4 z-[55]">
        <button 
          onClick={() => setShowSettings(!showSettings)} 
          className={`p-2 rounded-full transition-all duration-300 ${theme === 'dark' ? 'text-stone-500 hover:bg-stone-800' : 'text-stone-400 hover:bg-stone-200'}`}
        >
          <TextSizeIcon />
        </button>
        {showSettings && <SettingsDropdown {...settingsProps} />}
      </div>

      <div ref={contentRef} className="flex-grow overflow-y-auto px-6 py-4 hide-scrollbar pb-24"> 
        <AnimatePresence mode="wait">
          <motion.div
            key={step}
            initial={{ opacity: 0, y: 15, filter: "blur(8px)" }}
            animate={{ opacity: 1, y: 0, filter: "blur(0px)" }}
            exit={{ opacity: 0, y: -15, filter: "blur(8px)" }}
            transition={{ duration: 0.5, ease: [0.22, 1, 0.36, 1] }}
          >
            {children}
          </motion.div>
        </AnimatePresence>
      </div>

      <div className={`absolute bottom-0 left-0 w-full pt-8 pb-6 px-6 z-20 bg-gradient-to-t to-transparent ${theme === 'dark' ? 'from-[#1c1917] via-[#1c1917]' : theme === 'sepia' ? 'from-[#e3d5c6] via-[#e3d5c6]' : 'from-[#fffdf9] via-[#fffdf9]'}`}>
        {onNext && (
          <motion.button 
            onClick={onNext}
            whileHover={{ scale: 1.01 }}
            whileTap={{ scale: 0.98 }}
            className={`w-full py-4 rounded-2xl text-xs font-bold tracking-[0.25em] uppercase shadow-xl flex items-center justify-center gap-2 transition-all
              ${theme === 'dark' ? 'bg-stone-100 text-stone-900 hover:bg-stone-300' : 'bg-stone-900 text-stone-50 hover:bg-stone-800'}
            `}
          >
            {btnText} {step === 0 && <BookIcon />}
          </motion.button>
        )}
        
        <div className="flex justify-center space-x-2 mt-4">
            {[0, 1, 2, 3, 4].map((s) => (
                <div key={s} className={`h-1 rounded-full transition-all duration-700 
                  ${step === s 
                     ? (theme === 'dark' ? 'w-8 bg-stone-200' : 'w-8 bg-stone-800') 
                     : (theme === 'dark' ? 'w-1 bg-stone-700' : 'w-1 bg-stone-300')}
                `}/>
            ))}
        </div>
      </div>
    </div>
  );
};
